	mkdir -p $(BUILD_DIR)
	cmake -S . -B $(BUILD_DIR) $(CMAKE_FLAGS)

	cmake --build $(BUILD_DIR)

build: cmake_build

	cd $(BUILD_DIR) && ctest --output-on-failure
